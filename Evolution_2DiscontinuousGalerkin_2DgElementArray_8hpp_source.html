36 /*!
37  * \brief Import numeric initial data in the `ImportPhase`
38  *
39  * Requires the `InitialData` conforms to
40  * `evolution::protocols::NumericInitialData`. The `InitialData::import_fields`
41  * will be imported in the `ImportPhase` from the data file that is specified by
42  * the options in the `evolution::OptionTags::NumericInitialData` group.
43  *
44  * \see `DgElementArray`
45  */
46 template <typename PhaseType, PhaseType ImportPhase, typename InitialData>
47 struct ImportNumericInitialData {
48  static_assert(tt::assert_conforms_to<
49  InitialData, evolution::protocols::NumericInitialData>);
50  using phase_type = PhaseType;
51  static constexpr PhaseType import_phase = ImportPhase;
52  using initial_data = InitialData;
53 };

80 /*!
81  * \brief The parallel component responsible for managing the DG elements that
82  * compose the computational domain
83  *
84  * This parallel component will perform the actions specified by the
85  * `PhaseDepActionList`.
86  *
87  * This component also supports loading initial for its elements from a file.
88  * To do so, set the `ImportInitialData` template parameter to
89  * `ImportNumericInitialData`. See the documentation of the
90  * `ImportNumericInitialData` for details.
91  */
92 template <class Metavariables, class PhaseDepActionList,
93  class ImportInitialData = ImportNoInitialData>
94 struct DgElementArray {
95  static constexpr size_t volume_dim = Metavariables::volume_dim;
96 
97  using chare_type = Parallel::Algorithms::Array;
98  using metavariables = Metavariables;
99  using phase_dependent_action_list = PhaseDepActionList;
100  using array_index = ElementId<volume_dim>;
101 
102  using const_global_cache_tags =
103  tmpl::list<domain::Tags::Domain<volume_dim>,
104  typename DgElementArray_detail::import_numeric_data_cache_tags<
105  DgElementArray, ImportInitialData>::type>;
106 
107  using array_allocation_tags =
108  tmpl::list<domain::Tags::InitialRefinementLevels<volume_dim>>;
109 
110  using initialization_tags = Parallel::get_initialization_tags<
111  Parallel::get_initialization_actions_list<phase_dependent_action_list>,
112  array_allocation_tags>;
113 
114  static void allocate_array(
115  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache,
116  const tuples::tagged_tuple_from_typelist<initialization_tags>&
117  initialization_items) noexcept;
118 
119  static void execute_next_phase(
120  const typename Metavariables::Phase next_phase,
121  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache) noexcept {
122  auto& local_cache = *(global_cache.ckLocalBranch());
123  Parallel::get_parallel_component<DgElementArray>(local_cache)
124  .start_phase(next_phase);
125 
126  if constexpr (not std::is_same_v<ImportInitialData, ImportNoInitialData>) {
127  static_assert(
128  std::is_same_v<typename ImportInitialData::phase_type,
129  typename Metavariables::Phase>,
130  "Make sure the 'ImportNumericInitialData' type uses a 'Phase' "
131  "that is defined in the Metavariables.");
132  if (next_phase == ImportInitialData::import_phase) {
133  Parallel::threaded_action<
134  DgElementArray_detail::read_element_data_action<
135  DgElementArray, typename ImportInitialData::initial_data>>(
136  Parallel::get_parallel_component<
137  importers::VolumeDataReader<Metavariables>>(local_cache));
138  }
139  }
140  }
141 };

143 template <class Metavariables, class PhaseDepActionList,
144  class ImportInitialData>
145 void DgElementArray<Metavariables, PhaseDepActionList, ImportInitialData>::
146  allocate_array(
147  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache,
148  const tuples::tagged_tuple_from_typelist<initialization_tags>&
149  initialization_items) noexcept {
150  auto& local_cache = *(global_cache.ckLocalBranch());
151  auto& dg_element_array =
152  Parallel::get_parallel_component<DgElementArray>(local_cache);
153  const auto& domain =
154  Parallel::get<domain::Tags::Domain<volume_dim>>(local_cache);
155  const auto& initial_refinement_levels =
156  get<domain::Tags::InitialRefinementLevels<volume_dim>>(
157  initialization_items);
158  int which_proc = 0;
159  for (const auto& block : domain.blocks()) {
160  const auto initial_ref_levs = initial_refinement_levels[block.id()];
161  const std::vector<ElementId<volume_dim>> element_ids =
162  initial_element_ids(block.id(), initial_ref_levs);
163  const int number_of_procs = Parallel::number_of_procs();
164  for (size_t i = 0; i < element_ids.size(); ++i) {
165  dg_element_array(ElementId<volume_dim>(element_ids[i]))
166  .insert(global_cache, initialization_items, which_proc);
167  which_proc = which_proc + 1 == number_of_procs ? 0 : which_proc + 1;
168  }
169  }
170  dg_element_array.doneInserting();
171 }
