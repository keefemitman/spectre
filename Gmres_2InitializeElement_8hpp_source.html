25 template <typename FieldsTag, typename OptionsGroup, bool Preconditioned>
26 struct InitializeElement {
27  private:
28  using fields_tag = FieldsTag;
29  using initial_fields_tag =
30  db::add_tag_prefix<LinearSolver::Tags::Initial, fields_tag>;
31  using operator_applied_to_fields_tag =
32  db::add_tag_prefix<LinearSolver::Tags::OperatorAppliedTo, fields_tag>;
33  using operand_tag =
34  db::add_tag_prefix<LinearSolver::Tags::Operand, fields_tag>;
35  using preconditioned_operand_tag =
36  db::add_tag_prefix<LinearSolver::Tags::Preconditioned, operand_tag>;
37  using operator_applied_to_operand_tag = db::add_tag_prefix<
38  LinearSolver::Tags::OperatorAppliedTo,
39  std::conditional_t<Preconditioned, preconditioned_operand_tag,
40  operand_tag>>;
41  using orthogonalization_iteration_id_tag =
42  db::add_tag_prefix<LinearSolver::Tags::Orthogonalization,
43  LinearSolver::Tags::IterationId<OptionsGroup>>;
44  using basis_history_tag =
45  LinearSolver::Tags::KrylovSubspaceBasis<operand_tag>;
46 
47  public:
48  template <typename DbTagsList, typename... InboxTags, typename Metavariables,
49  typename ArrayIndex, typename ActionList,
50  typename ParallelComponent>
51  static auto apply(db::DataBox<DbTagsList>& box,
52  const tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
53  Parallel::ConstGlobalCache<Metavariables>& /*cache*/,
54  const ArrayIndex& /*array_index*/,
55  const ActionList /*meta*/,
56  const ParallelComponent* const /*meta*/) noexcept {
57  auto initial_box = ::Initialization::merge_into_databox<
58  InitializeElement,
59  db::AddSimpleTags<LinearSolver::Tags::IterationId<OptionsGroup>,
60  initial_fields_tag, operator_applied_to_fields_tag,
61  operand_tag, operator_applied_to_operand_tag,
62  orthogonalization_iteration_id_tag, basis_history_tag,
63  LinearSolver::Tags::HasConverged<OptionsGroup>>>(
64  std::move(box),
65  // The `PrepareSolve` action populates these tags with initial values,
66  // except for `operator_applied_to_fields_tag` which is expected to be
67  // filled at that point and `operator_applied_to_operand_tag` which is
68  // expected to be updated in every iteration of the algorithm.
69  std::numeric_limits<size_t>::max(), db::item_type<initial_fields_tag>{},
70  db::item_type<operator_applied_to_fields_tag>{},
71  db::item_type<operand_tag>{},
72  db::item_type<operator_applied_to_operand_tag>{},
73  std::numeric_limits<size_t>::max(), db::item_type<basis_history_tag>{},
74  Convergence::HasConverged{});
75 
76  if constexpr (not Preconditioned) {
77  return std::make_tuple(std::move(initial_box));
78  } else {
79  using preconditioned_basis_history_tag =
80  LinearSolver::Tags::KrylovSubspaceBasis<preconditioned_operand_tag>;
81 
82  return std::make_tuple(::Initialization::merge_into_databox<
83  InitializeElement,
84  db::AddSimpleTags<preconditioned_basis_history_tag,
85  preconditioned_operand_tag>>(
86  std::move(initial_box),
87  db::item_type<preconditioned_basis_history_tag>{},
88  db::item_type<preconditioned_operand_tag>{}));
89  }
90  }
91 };
