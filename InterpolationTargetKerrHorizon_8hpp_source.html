121 /// \brief Computes points on a Kerr horizon.
122 ///
123 /// The points are such that they conform to the horizon of a Kerr
124 /// black hole (in Kerr-Schild coordinates) with given center, mass,
125 /// and dimensionless spin, as specified in the options.
126 ///
127 /// \note The returned points are not actually on the horizon;
128 /// instead, they are collocation points of a Strahlkorper whose
129 /// spectral representation matches the horizon shape up to order
130 /// Lmax, where Lmax is the spherical-harmonic order specified in the
131 /// options. As Lmax increases, the returned points converge to the
132 /// horizon.
133 ///
134 /// For requirements on InterpolationTargetTag, see InterpolationTarget
135 template <typename InterpolationTargetTag, typename Frame>
136 struct KerrHorizon {
137  using const_global_cache_tags =
138  tmpl::list<Tags::KerrHorizon<InterpolationTargetTag>>;
139  using initialization_tags =
140  tmpl::append<StrahlkorperTags::items_tags<Frame>,
141  StrahlkorperTags::compute_items_tags<Frame>>;
142  using is_sequential = std::false_type;
143  template <typename DbTags, typename Metavariables>
144  static auto initialize(
145  db::DataBox<DbTags>&& box,
146  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
147  const auto& options =
148  Parallel::get<Tags::KerrHorizon<InterpolationTargetTag>>(cache);
149 
150  // Make a Strahlkorper with the correct shape.
151  ::Strahlkorper<Frame> strahlkorper(
152  options.l_max, options.l_max,
153  get(gr::Solutions::kerr_horizon_radius(
154  ::YlmSpherepack(options.l_max, options.l_max).theta_phi_points(),
155  options.mass, options.dimensionless_spin)),
156  options.center);
157 
158  // Put Strahlkorper and its ComputeItems into a new DataBox.
159  return db::create_from<
160  db::RemoveTags<>,
161  db::AddSimpleTags<StrahlkorperTags::items_tags<Frame>>,
162  db::AddComputeTags<StrahlkorperTags::compute_items_tags<Frame>>>(
163  std::move(box), std::move(strahlkorper));
164  }

166  template <typename Metavariables, typename DbTags, typename TemporalId>
167  static tnsr::I<DataVector, 3, ::Frame::Inertial> points(
168  const db::DataBox<DbTags>& box,
169  const tmpl::type_<Metavariables>& /*meta*/,
170  const TemporalId& /*temporal_id*/) noexcept {
171  // In the future, when we add support for multiple Frames,
172  // the code that transforms coordinates from the Strahlkorper Frame
173  // to Frame::Inertial will go here. That transformation
174  // may depend on `temporal_id`.
175  return db::get<StrahlkorperTags::CartesianCoords<::Frame::Inertial>>(box);
176  }

178  static tnsr::I<DataVector, 3, ::Frame::Inertial> points(
179  const db::DataBox<DbTags>& box,
180  const tmpl::type_<Metavariables>& /*meta*/) noexcept {
181  return db::get<StrahlkorperTags::CartesianCoords<::Frame::Inertial>>(box);
182  }
