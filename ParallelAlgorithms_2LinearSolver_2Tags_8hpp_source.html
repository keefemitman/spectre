106 /// Compute the residual \f$r=b - Ax\f$ from the `SourceTag` \f$b\f$ and the
107 /// `db::add_tag_prefix<LinearSolver::Tags::OperatorAppliedTo, FieldsTag>`
108 /// \f$Ax\f$.
109 template <typename FieldsTag, typename SourceTag>
110 struct ResidualCompute : db::add_tag_prefix<Residual, FieldsTag>,
111  db::ComputeTag {
112  using base = db::add_tag_prefix<Residual, FieldsTag>;

114  tmpl::list<SourceTag, db::add_tag_prefix<OperatorAppliedTo, FieldsTag>>;
115  using return_type = typename base::type;
116  static void function(
117  const gsl::not_null<return_type*> residual,
118  const typename SourceTag::type& source,
119  const db::item_type<db::add_tag_prefix<OperatorAppliedTo, FieldsTag>>&
120  operator_applied_to_fields) noexcept {
121  *residual = source - operator_applied_to_fields;

332 /*!
333  * \brief `Convergence::Criteria` that determine the linear solve has converged
334  *
335  * \note The smallest possible residual magnitude the linear solver can reach is
336  * the product between the machine epsilon and the condition number of the
337  * linear operator that is being inverted. Smaller residuals are numerical
338  * artifacts. Requiring an absolute or relative residual below this limit will
339  * likely lead to termination by `MaxIterations`.
340  *
341  * \note Remember that when the linear operator \f$A\f$ corresponds to a PDE
342  * discretization, decreasing the linear solver residual below the
343  * discretization error will not improve the numerical solution any further.
344  * I.e. the error \f$e_k=x_k-x_\mathrm{analytic}\f$ to an analytic solution
345  * will be dominated by the linear solver residual at first, but even if the
346  * discretization \f$Ax_k=b\f$ was exactly solved after some iteration \f$k\f$,
347  * the discretization residual
348  * \f$Ae_k=b-Ax_\mathrm{analytic}=r_\mathrm{discretization}\f$ would still
349  * remain. Therefore, ideally choose the absolute or relative residual criteria
350  * based on an estimate of the discretization residual.
351  */
352 template <typename OptionsGroup>
353 struct ConvergenceCriteria : db::SimpleTag {
354  static std::string name() noexcept {
355  return "ConvergenceCriteria(" + option_name<OptionsGroup>() + ")";
356  }
357  using type = Convergence::Criteria;
358  using option_tags =
359  tmpl::list<LinearSolver::OptionTags::ConvergenceCriteria<OptionsGroup>>;
360 
361  static constexpr bool pass_metavariables = false;
362  static Convergence::Criteria create_from_options(
363  const Convergence::Criteria& convergence_criteria) noexcept {
364  return convergence_criteria;
365  }
366 };
